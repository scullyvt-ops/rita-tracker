const SHEET_NAME = "Readings"; const COLUMNS = [ "ID", "Timestamp", "Glucose", "Status", "Called Homestead", "Call Time", "Intervention 1", "Qty 1", "Intervention 2", "Qty 2", "Intervention 3", "Qty 3", "Notes", "Screenshots #", "Last Updated" ]; function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(COLUMNS); const hdr = sheet.getRange(1, 1, 1, COLUMNS.length); hdr.setBackground("#2D5A3D"); hdr.setFontColor("white"); hdr.setFontWeight("bold"); sheet.setFrozenRows(1); [80,160,80,80,140,140,130,100,130,100,130,100,200,100,160] .forEach(function(w, i) { sheet.setColumnWidth(i + 1, w); }); } return sheet; } function rowFromEntry(entry) { var invs = entry.interventions || []; return [ String(entry.id || ""), entry.timestamp || "", Number(entry.glucose)|| 0, entry.status || "", entry.calledHomestead ? "Yes" : "No", entry.calledTime || "", invs[0] ? invs[0].name : "", invs[0] ? invs[0].quantity : "", invs[1] ? invs[1].name : "", invs[1] ? invs[1].quantity : "", invs[2] ? invs[2].name : "", invs[2] ? invs[2].quantity : "", entry.notes || "", Number(entry.screenshotCount) || 0, new Date().toISOString() ]; } function entryFromRow(row) { if (!row[0]) return null; var invs = []; if (row[6]) invs.push({ name: String(row[6]), quantity: String(row[7] || "") }); if (row[8]) invs.push({ name: String(row[8]), quantity: String(row[9] || "") }); if (row[10]) invs.push({ name: String(row[10]), quantity: String(row[11] || "") }); return { id: String(row[0]), timestamp: String(row[1] || ""), glucose: Number(row[2]) || 0, status: String(row[3] || "NORMAL"), calledHomestead: String(row[4]) === "Yes", calledTime: row[5] ? String(row[5]) : null, interventions: invs, notes: String(row[12] || ""), screenshotCount: Number(row[13]) || 0, screenshots: [] }; } function respond(data, callback) { var json = JSON.stringify(data); if (callback) { return ContentService .createTextOutput(callback + "(" + json + ")") .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { try { var sheet = getSheet(); var cb = (e.parameter && e.parameter.callback) ? e.parameter.callback : null; if (e.parameter && e.parameter.payload) { var body = JSON.parse(e.parameter.payload); var action = body.action; if (action === "save") { var entry = body.entry; var rows = sheet.getDataRange().getValues(); var found = -1; for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(entry.id)) { found = i + 1; break; } } var rowData = rowFromEntry(entry); if (found > 0) { sheet.getRange(found, 1, 1, COLUMNS.length).setValues([rowData]); } else { sheet.appendRow(rowData); } return respond({ ok: true, action: "saved", id: entry.id }, cb); } if (action === "delete") { var id = String(body.id); var rows = sheet.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (String(rows[i][0]) === id) { sheet.deleteRow(i + 1); return respond({ ok: true, action: "deleted", id: id }, cb); } } return respond({ ok: false, error: "ID not found: " + id }, cb); } return respond({ ok: false, error: "Unknown action: " + action }, cb); } var rows = sheet.getDataRange().getValues(); var entries = []; for (var i = 1; i < rows.length; i++) { var entry = entryFromRow(rows[i]); if (entry) entries.push(entry); } return respond({ ok: true, entries: entries }, cb); } catch (err) { return respond({ ok: false, error: err.message }, cb); } } function doPost(e) { return doGet(e); }
